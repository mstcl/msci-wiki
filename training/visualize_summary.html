j = 4
iteration = 500

plot_configs = get_plot_configs()
sns.set(rc=plot_configs)
sns.set_style("ticks")
sns.set_style({"xtick.direction": "in","ytick.direction": "in"})
plt.rcParams.update(plot_configs)

fig = plt.figure(figsize=(8,6), constrained_layout=True)
gspec = fig.add_gridspec(3, 4)


model = tf.keras.models.load_model('models/blue9392.keras')
model_mini = Model(inputs=model.inputs, outputs=model.layers[2].output)

img = hf[f"conf_{iters[iteration]}"]
img = np.array(img)

feature_maps1 = model_mini.predict(np.expand_dims(img.reshape((img.shape[0], img.shape[1], 1)), axis=0), verbose=0)

for jdx in range(j):
    ax = fig.add_subplot(gspec[0, jdx])
    ax.set_xticks([])
    ax.set_yticks([])
    ax.text(s=f"A{jdx+1}", x=0.05, y=0.85, transform=ax.transAxes, backgroundcolor=(1,1,1,0.9))
    if jdx == 0:
        plt.imshow(img, cmap='cubehelix')
    else:
        plt.imshow(feature_maps1[0, :, :, jdx-1], cmap='cubehelix')
