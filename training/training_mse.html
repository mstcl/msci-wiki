acc1 = history_mse['mse']
val_acc1 = history_mse['val_mse']
acc2 = history_mae['mae']
val_acc2 = history_mae['val_mae']

df = pd.DataFrame(
    {
        "Training": np.concatenate((acc1,acc2)),
        "Validation": np.concatenate((val_acc1,val_acc2)),
        "Loss function": ["MSE"]*100 + ["MAE"]*100
    }
)

df_pred = pd.DataFrame(
    {
        "Predictions": np.concatenate((prediction_mse.T[0]-y_val,prediction_mae.T[0]-y_val)),
        "Actual": np.tile(y_val,2),
        "Loss function": ["MSE"]*6000+["MAE"]*6000
    }
)

palette={"MAE": mcm.gnuplot(150), "MSE": mcm.gnuplot(30)}
epochs_range = np.tile(range(100),2)
fig, ax1 = plt.subplots(1,1, figsize=(10, 7))
sns.lineplot(ax=ax1, data=df, x=epochs_range, y="Training", hue='Loss function', palette=palette, legend=False)
sns.lineplot(ax=ax1, data=df, x=epochs_range, y="Validation", hue='Loss function', legend=False, ls='--', palette=palette)
ax1.set_ylim(bottom=-0.015)

ax1.annotate(text='Training MAE', xytext=(50,0.045), xy=(50,0.032), c=palette["MAE"], ha="center",
             arrowprops={"arrowstyle": "-", "color": palette["MAE"]})
ax1.annotate(text='Validation MAE', xytext=(85,0.045), xy=(85,0.014), c=palette["MAE"], ha="center",
             arrowprops={"arrowstyle": "-", "color": palette["MAE"]})

ax1.annotate(text='Training MSE', xytext=(15,0.02), xy=(15,0.009), c=palette["MSE"], ha="center",
             arrowprops={"arrowstyle": "-", "color": palette["MSE"]})
ax1.annotate(text='Validation MSE', xytext=(10,-0.005), xy=(10,0.012), c=palette["MSE"], ha="center",
             arrowprops={"arrowstyle": "-", "color": palette["MSE"]})

ax1.set_ylabel("Loss")
ax1.set_xlabel("Training epochs")

ax = inset_axes(ax1, width="100%", height="100%", borderpad=1)
ax.set_axes_locator(InsetPosition(ax1, [0.38, 0.5, 0.6, 0.46]))

sns.violinplot(
    ax=ax,
    data=df_pred,
    x="Actual",
    y="Predictions",
    hue="Loss function",
    color="w",
    alpha=.9,
    density_norm="width",
    palette=palette,
    native_scale=True,
    log_scale=(True,False),
    linewidth=1,
    inner="box",
    inner_kws={"box_width": 1, "whis_width": 0, "color": "w", "marker": None},
)

ax.get_xaxis().set_major_formatter(ticker.ScalarFormatter())

ax.axhline(0, zorder=0, c='.75', ls="--")
sns.move_legend(ax, "lower left", frameon=False, fontsize=14)
