    {
        "Predicted": np.concatenate(
            (
                predictions_with_orientation,
                predictions_without_orientation,
                predictions_with_monochrome,
            )
        ),
        "Actual": np.concatenate((actual_with_orientation, actual_without_orientation, actual_with_monochrome)),
        "Context": pd.Categorical(
            ["Trained with orientation"] * len(actual_with_orientation)
            + ["Trained without orientation"] * len(actual_without_orientation)
            + ["Trained with thresholding"] * len(actual_with_monochrome),
            categories=["Trained with orientation", "Trained without orientation", "Trained with thresholding"],
        ),
    }
)

df2 = pd.DataFrame(
    {
        "STD": np.concatenate(
            (
                df.query('Context == "Trained with orientation"')
                .groupby("Actual", as_index=False)["Predicted"]
                .std()["Predicted"],
                df.query('Context == "Trained without orientation"')
                .groupby("Actual", as_index=False)["Predicted"]
                .std()["Predicted"],
                df.query('Context == "Trained with thresholding"')
                .groupby("Actual", as_index=False)["Predicted"]
                .std()["Predicted"],
            )
        ),
        "Mean": np.concatenate(
            (
                df.query('Context == "Trained with orientation"')
                .groupby("Actual", as_index=False)["Predicted"]
                .mean()["Predicted"],
                df.query('Context == "Trained without orientation"')
                .groupby("Actual", as_index=False)["Predicted"]
                .mean()["Predicted"],
                df.query('Context == "Trained with thresholding"')
                .groupby("Actual", as_index=False)["Predicted"]
                .mean()["Predicted"],
            )
        ),
        "Actual": np.tile(np.logspace(-6, -1, 10, base=2), 3),
        "Context": pd.Categorical(
