import seaborn as sns
import numpy as np
import pandas as pd
import matplotlib.pyplot as plt
import matplotlib.ticker as ticker

tumbles = np.logspace(-6,-1,10,base=2)
plot_configs = {
        "axes.formatter.use_mathtext": True,
        "font.size": 14,
        "axes.titlesize": 14,
        "axes.labelsize": 16,
        "xtick.labelsize": 14,
        "ytick.labelsize": 14,
        "legend.fontsize": 16,
        "font.family": "sans-serif",
        "font.sans-serif": "Arial",
        "mathtext.fontset": "cm",
        "text.usetex": False,
}
sns.set(rc=plot_configs)
sns.set_style("ticks")
sns.set_style({"xtick.direction": "in", "ytick.direction": "in"})
plt.rcParams.update(plot_configs)

df = pd.read_csv('hyphen1065.csv')

fig, ax = plt.subplots(figsize=(6, 6), constrained_layout=True)
ax.scatter(
    data=df,
    x="Actual values",
    y="Predicted values",
    marker='o',
    fc="darkblue",
    alpha=0.1,
)
ax.scatter(data=df,x='Actual values',y='Actual values', marker='o', s=100, fc="darkorange", ec='k')
ax.legend(frameon=False, fontsize='medium')
ax.set_xscale('log')
ax.get_xaxis().set_major_formatter(ticker.ScalarFormatter())
ax.set_xticks(np.round(tumbles,3)[::2])
ax.set_yscale('log')
ax.get_yaxis().set_major_formatter(ticker.ScalarFormatter())
ax.set_yticks(np.round(tumbles,3)[::2])
ax.set(xlabel=r"Actual tumbling rate, $\alpha_a$", ylabel=r"Predicted tumbling rate, $\alpha_p$")
sns.despine()
