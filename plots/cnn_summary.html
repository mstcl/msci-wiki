IQR changes when going to higher packing fraction

Code 
df_before = pd.read_csv("cache/model_summary.csv")
df_after = pd.read_csv("cache/model_summary_hi.csv")
df_analysis = pd.DataFrame(
    {
        "IQR changes": np.concatenate(
            (
                (
                    df_after.query('Context == "Trained with orientation"')
                    .groupby("Actual", as_index=False)["Predicted"]
                    .agg(iqr)["Predicted"]
                    - df_before.query('Context == "Trained with orientation"')
                    .groupby("Actual", as_index=False)["Predicted"]
                    .agg(iqr)["Predicted"]
                )
                / df_before.query('Context == "Trained with orientation"')
                .groupby("Actual", as_index=False)["Predicted"]
                .agg(iqr)["Predicted"],
                (
                    df_after.query('Context == "Trained without orientation"')
                    .groupby("Actual", as_index=False)["Predicted"]
                    .agg(iqr)["Predicted"]
                    - df_before.query('Context == "Trained without orientation"')
                    .groupby("Actual", as_index=False)["Predicted"]
                    .agg(iqr)["Predicted"]
                )
                / df_before.query('Context == "Trained without orientation"')
                .groupby("Actual", as_index=False)["Predicted"]
                .agg(iqr)["Predicted"],
                (
                    df_after.query('Context == "Trained with thresholding"')
                    .groupby("Actual", as_index=False)["Predicted"]
                    .agg(iqr)["Predicted"]
                    - df_before.query('Context == "Trained with thresholding"')
                    .groupby("Actual", as_index=False)["Predicted"]
                    .agg(iqr)["Predicted"]
                )
                / df_before.query('Context == "Trained with thresholding"')
                .groupby("Actual", as_index=False)["Predicted"]
                .agg(iqr)["Predicted"],
            )
        ),
        "Actual": np.tile(np.logspace(-6, -1, 10, base=2), 3),
        "Context": (
            ["Trained with orientation"] * 10
            + ["Trained without orientation"] * 10
            + ["Trained with thresholding"] * 10,
        ),
    }
)
