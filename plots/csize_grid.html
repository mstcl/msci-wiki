def save_points():
    densities = [0.15,0.25,0.45]
    tumbles = [0.016,0.073,0.340]

    sizes = []
    ds = []
    ts = []
    
    for idx, d in enumerate(densities):
        for jdx, t in enumerate(tumbles):
            file = f"no_roll_data/dataset_tumble_{t:.3f}_{d}.h5"
            all_cs = np.array([])
            for idx2 in range(500,1000,1):
                labelled, nlabels = get_cluster_labels(file, idx2)
                lb = labelled.flatten()
                cs = np.bincount(labelled.flatten())[1:]
                all_cs = np.hstack((all_cs,cs))
            ds.append(d)
            ts.append(t)
            sizes.append(all_cs.tolist())
    csv_out = pd.DataFrame()
    csv_out.insert(0, "alphas", ts)
    csv_out.insert(1, "phis", ds)
    csv_out.insert(2, "sizes", sizes)
    csv_out.to_csv(f"cache/cluster_distribution.csv")

densities = [0.15,0.25,0.45]

            alpha, rho
        )
        hf = h5py.File(f"no_roll_data/dataset_tumble_{alpha:.3f}_{rho}.h5", "r")
        key_list = list(hf.keys())
        iters = get_ds_iters(key_list)
        img = hf[f"conf_{iters[-1]}"]
        axins.matshow(img, cmap=cmap)
        axins.set_yticks([])
        axins.set_xticks([])
        axis.scatter(
            x=bin_edges[1:],
            y=count,
            edgecolor=cm.batlow(40, alpha=1),
            facecolor=cm.batlow(40, alpha=0.2),
            zorder=2
        )
        axis.set_axisbelow(True)
        axis.set_ylim((0.8*10**(-9), 4*10**1))
        axis.set_xlim(x_lim)
        axis.set_yscale("log"), axis.set_xscale("log")
        axis.text(

def func(x, a, b, c):
