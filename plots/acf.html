def overlap(traj,i,j):
    return ((traj[i]>0)*(traj[j]>0)).mean()

tumbles = np.logspace(-6, -1, 10, base=2)
lags = np.arange(0, 100 + 1, 1)

acfs = []

for t in tumbles:
    hf = h5py.File(f'no_roll_data/dataset_tumble_{t:.3f}_0.25.h5')
    data = [ hf[i][:] for i in natsort.natsorted(list(hf.keys()))]
    data = np.array(data)
    acf =[]
    # moving window size 10, 50 times
    for i in range(0, 500, 10):
        a = [overlap(data,i,i+lag) for lag in lags]
        acf.append(a)
    acf = np.asarray(acf).mean(axis=0) # mean across all lag
    #acf = acf-acf[-1]

    acfs.append(acf)
    ts.append(np.repeat(t,len(lags)))
